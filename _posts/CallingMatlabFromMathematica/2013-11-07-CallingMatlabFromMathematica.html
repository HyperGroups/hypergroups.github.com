{% raw %}

matrix = RotationMatrix[90 Degree, {0, 0, 1}]
{% endraw %}

{% raw %}

matrix2Matlab[matrix_] := 

 "[" <> StringReplace[

   ExportString[matrix, "CSV"], {"\n" -> ";", "," -> " "}] <> "]"
{% endraw %}

{% raw %}

matrixMatlab = matrix2Matlab[matrix]
{% endraw %}

{% raw %}

MEvaluate[

 "[yaw, pitch, roll] = dcm2angle(" <> matrixMatlab <> ",'YXZ')"]
{% endraw %}
